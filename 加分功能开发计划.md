```python
"""
大模型服务 - 生成个性化健康建议
"""
import dashscope
from dashscope import Generation
from typing import Dict, List
import os
from app.core.config import settings

dashscope.api_key = settings.DASHSCOPE_API_KEY


class LLMService:
    """大语言模型服务"""
    
    def __init__(self):
        self.model = "qwen-turbo"  # 或 qwen-plus (效果更好但慢一点)
    
    def generate_personalized_recommendations(
        self, 
        user_data: Dict, 
        risk_result: Dict
    ) -> str:
        """
        生成个性化健康建议
        
        Args:
            user_data: 用户问卷数据
            risk_result: 风险评估结果
            
        Returns:
            str: AI生成的个性化建议（200-300字）
        """
        # 构建提示词
        prompt = self._build_prompt(user_data, risk_result)
        
        # 调用大模型
        response = Generation.call(
            model=self.model,
            prompt=prompt,
            max_tokens=500,
            temperature=0.7,  # 创造性（0-1）
            top_p=0.8
        )
        
        if response.status_code == 200:
            return response.output.text.strip()
        else:
            # 降级处理：返回规则生成的建议
            return self._fallback_recommendation(risk_result)
    
    def _build_prompt(self, user_data: Dict, risk_result: Dict) -> str:
        """构建Prompt"""
        
        # 提取关键信息
        age = user_data.get('age')
        gender = user_data.get('gender')
        bmi = user_data.get('bmi', 0)
        risk_score = risk_result['overall_risk']['score']
        risk_level = risk_result['overall_risk']['level']
        key_factors = risk_result['key_factors'][:3]  # 前3个关键因素
        
        # 构建因素列表
        factors_str = "\n".join([
            f"- {f['factor']}：{f['description']}" 
            for f in key_factors
        ])
        
        prompt = f"""你是一位专业、温和的肿瘤预防医生。请根据以下患者信息，生成个性化的健康建议。

【患者信息】
- 年龄：{age}岁
- 性别：{gender}
- BMI：{bmi:.1f}
- 综合风险评分：{risk_score*100:.0f}分（{risk_level}）

【主要风险因素】
{factors_str}

【要求】
1. 语气温和、专业，避免引起恐慌
2. 重点针对可改变的风险因素给出建议
3. 给出3-4条具体、可操作的建议
4. 包含生活方式、饮食、筛查等方面
5. 字数控制在200-300字
6. 如果是高风险，建议就医但不要过度夸大
7. 使用简洁的段落，不要使用编号列表

请直接生成建议内容，不要有多余的开场白或总结。"""

        return prompt
    
    def _fallback_recommendation(self, risk_result: Dict) -> str:
        """降级方案：规则生成建议"""
        risk_level = risk_result['overall_risk']['level']
        
        templates = {
            "低风险": "您目前的肿瘤风险较低，建议继续保持良好的生活习惯。定期体检，关注身体变化。",
            "中风险": "您的肿瘤风险处于中等水平，建议调整生活方式，增加体检频率。",
            "高风险": "您的肿瘤风险较高，建议尽快到医院进行全面体检，咨询专业医生。",
            "极高风险": "您存在多项高危因素，强烈建议立即就医，进行全面检查。"
        }
        
        return templates.get(risk_level, "建议定期体检，关注健康。")
    
    def generate_comparison_summary(
        self, 
        old_assessment: Dict, 
        new_assessment: Dict
    ) -> str:
        """
        生成历史对比总结（可选功能）
        
        Args:
            old_assessment: 旧评估
            new_assessment: 新评估
            
        Returns:
            str: AI生成的对比分析
        """
        old_score = old_assessment['overall_risk']['score']
        new_score = new_assessment['overall_risk']['score']
        change = (new_score - old_score) * 100
        
        prompt = f"""作为健康顾问，分析患者两次肿瘤筛查的变化：

第一次评估（{old_assessment['created_at']}）：
- 风险分数：{old_score*100:.0f}分

第二次评估（{new_assessment['created_at']}）：
- 风险分数：{new_score*100:.0f}分
- 变化：{'上升' if change > 0 else '下降'}{abs(change):.0f}分

请用100字内总结变化情况，并给出鼓励或建议。语气温和积极。"""

        response = Generation.call(
            model=self.model,
            prompt=prompt,
            max_tokens=200
        )
        
        if response.status_code == 200:
            return response.output.text.strip()
        else:
            return f"您的风险分数{'上升' if change > 0 else '下降'}了{abs(change):.0f}分。"


# 全局实例
llm_service = LLMService()
```

```python
"""
知识图谱API
"""
from fastapi import APIRouter, Query
from typing import Optional
import json
import os

router = APIRouter()


def load_knowledge_graph() -> dict:
    """加载知识图谱数据"""
    kg_path = os.path.join("data", "knowledge_graph.json")
    with open(kg_path, 'r', encoding='utf-8') as f:
        return json.load(f)


@router.get("/graph", summary="获取完整知识图谱")
async def get_knowledge_graph():
    """
    获取完整知识图谱数据
    
    Returns:
        包含nodes和edges的图数据
    """
    return load_knowledge_graph()


@router.get("/subgraph", summary="获取子图")
async def get_subgraph(
    center_node: str = Query(..., description="中心节点ID"),
    depth: int = Query(1, ge=1, le=3, description="扩展深度")
):
    """
    获取以某节点为中心的子图
    
    Args:
        center_node: 中心节点ID（如 "cancer_lung"）
        depth: 扩展深度（1-3跳）
        
    Returns:
        子图数据
    """
    kg = load_knowledge_graph()
    
    # 找到相关节点和边
    related_nodes = set([center_node])
    related_edges = []
    
    for _ in range(depth):
        new_nodes = set()
        for edge in kg['edges']:
            if edge['source'] in related_nodes:
                new_nodes.add(edge['target'])
                related_edges.append(edge)
            elif edge['target'] in related_nodes:
                new_nodes.add(edge['source'])
                related_edges.append(edge)
        related_nodes.update(new_nodes)
    
    # 筛选节点
    nodes = [n for n in kg['nodes'] if n['id'] in related_nodes]
    
    return {
        "nodes": nodes,
        "edges": related_edges,
        "center": center_node
    }


@router.get("/path", summary="查询两节点间路径")
async def find_path(
    start: str = Query(..., description="起始节点"),
    end: str = Query(..., description="终止节点"),
    max_depth: int = Query(3, description="最大搜索深度")
):
    """
    查找两个节点之间的关系路径
    
    示例：
        /path?start=risk_smoking&end=cancer_lung
        返回：吸烟 -> 肺癌 的关系路径
    """
    kg = load_knowledge_graph()
    
    # 简单的BFS路径查找
    from collections import deque
    
    queue = deque([(start, [start])])
    visited = set([start])
    
    while queue:
        node, path = queue.popleft()
        
        if len(path) > max_depth:
            continue
            
        if node == end:
            # 找到路径，构建详细信息
            path_details = []
            for i in range(len(path) - 1):
                # 找到连接边
                edge = next(
                    (e for e in kg['edges'] 
                     if (e['source'] == path[i] and e['target'] == path[i+1]) or
                        (e['target'] == path[i] and e['source'] == path[i+1])),
                    None
                )
                if edge:
                    path_details.append({
                        'from': path[i],
                        'to': path[i+1],
                        'relation': edge['relation']
                    })
            
            return {
                "found": True,
                "path": path,
                "details": path_details
            }
        
        # 扩展邻居节点
        for edge in kg['edges']:
            next_node = None
            if edge['source'] == node:
                next_node = edge['target']
            elif edge['target'] == node:
                next_node = edge['source']
            
            if next_node and next_node not in visited:
                visited.add(next_node)
                queue.append((next_node, path + [next_node]))
    
    return {
        "found": False,
        "message": "未找到路径"
    }
```
